            ul_zakladki_dynamiczne = '#zakladki-dynamiczne',

            $(ul_zakladki_dynamiczne+' li').addClass('nieaktywna');
            $(ul_linki_dynamiczne+' li').addClass('nieaktywna');
            $(ul_linki_dynamiczne+' li:not(:last-child)').removeClass('ostatnia');

        var obsluzOstatniLink = function(kliknieta_zakladka) {
            if ($(ul_linki_dynamiczne+' li').has('a[href='+kliknieta_zakladka+']').hasClass('ostatnia')) {
                $(ul_linki_dynamiczne+' li.nieaktywna:last').addClass('ostatnia');

        var obluzKlikZakladka = function(zakladka) {
            var kliknieta_zakladka = $('a', $(zakladka)).attr('href');
            ukryjZakladkiOrazLInki();
            $(zakladka).removeClass('nieaktywna');
            $(ul_linki_dynamiczne+' li').has('a[href='+kliknieta_zakladka+']').removeClass('nieaktywna');

            $(kliknieta_zakladka).removeClass('schowaj');

                        $(ul_linki_dynamiczne+' li').has('a[href='+kliknieta_zakladka+']').removeClass('nieaktywna');
                        obsluzOstatniLink(kliknieta_zakladka);
                        $(zakladki).addClass('schowaj');
                        $(kliknieta_zakladka).removeClass('schowaj');
                    });
            }
        }

        $(ul_zakladki_dynamiczne+' li').click(function() {
             obluzKlikZakladka(this);
