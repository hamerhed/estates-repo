            GR('oTrackEvents', [{elements: gaLink.toArray(), href: adres, value: wartosc}]);
            gaLink.click();
        }

        var ustawCiachoPozycji = function (nazwa, wartosc) {
            GR('ustawCookie', nazwa, wartosc, 60, '/');
        }

        var dopelnijZerami = function(wartosc) {
            if (wartosc < maksymalnyNumerZajawki) {
                wartosc = ('000'+wartosc).slice(-3);
            }
            return wartosc;
        }

        var obliczPozycje = function(element) {
            return dopelnijZerami(offset*limit + element.data('pozycja'));
        };

        var obsluzKlikniecia = function() {
            $('#lista-ogloszen li.linkDoKarty').each(function () {
                $(this).one('click', function() {
                    if (obliczPozycje($(this)) <= maksymalnyNumerZajawki) {
                        wyslijTracker(
                            '/pozycja/'+obliczPozycje($(this))+'|'+nazwaKategorii+'|'+region+'|'+miejscowosc+'/lista-klik/',
                            $(this).data('punkty-wyroznienia')
                        );
                        ustawCiachoPozycji(
                            idKategorii+'-'+$(this).attr('id').replace('lista-wiersz-', ''),
                            '/pozycja/'+obliczPozycje($(this))+'|'+nazwaKategorii+'|'+region+'|'+miejscowosc
                        );
                    };
               });
            });
        };

        /**********************************/
        /****** wykonywanie funkcji *******/
        /**********************************/
        obsluzKlikniecia();


  
        /**********************************/
        /****** monitoring pozycji ********/
        /**********************************/
        var pozycje_stare = [];

        function monitorujPozycje() {
            var pozycje_widoczne = [],
                pozycje_nowe = [];

            $('.linkDoKarty').each(function(){
                if ($(this).visible(true)) {
                    var pozycja = $(this).attr('data-pozycja');
                    if (pozycja) {
                        var pozycja_dane = {},
                            aktualna_pozycja = offset * limit + parseInt(pozycja);
                        if (aktualna_pozycja <= maksymalnyNumerZajawki) {
                            pozycja_dane.pozycja = aktualna_pozycja;
                            pozycja_dane.punkty = parseInt($(this).attr('data-punkty-wyroznienia'));
                            pozycje_widoczne[aktualna_pozycja] = pozycja_dane;
                        }
                   }
                }
            });

            $(pozycje_widoczne).each(function(klucz, wartosc){
                if (wartosc && $.inArray(klucz, pozycje_stare) == -1) {
                    pozycje_nowe.push(wartosc);
                    pozycje_stare.push(klucz);
                }
            });

            if (pozycje_nowe.length) {
                var dane = {};
                dane.id_kategoria = idKategorii;
                dane.id_region = id_region;
                dane.miejscowosc = miejscowosc;
                dane.zajawki = pozycje_nowe;
                if (!$.isEmptyObject(dane)) {
                    $.get('/dom-statystyki/lista/', dane);
                }
            }
         }

        setInterval(monitorujPozycje, 500);

  
    });
})(jQuery);
